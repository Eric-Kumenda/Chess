## ✅ Data Preprocessing Pipeline

### 📦 1. Filter Dataset for Winning Moves

```python
import pandas as pd

df = pd.read_csv("your_file.csv")
df = df.dropna(subset=["moves", "winner"])

# Tokenize moves
df['move_list'] = df['moves'].apply(lambda x: x.strip().split())

# Keep only rows where player made move and eventually won
def generate_training_rows(row):
    moves = row['move_list']
    winner = row['winner']
    training_rows = []

    # Determine who played each move
    for i in range(4, len(moves) - 1):  # minimum 4 previous moves
        side = 'white' if i % 2 == 0 else 'black'
        if side == winner:
            history = moves[i-4:i]  # last 4 moves before current
            next_move = moves[i]
            player_turn = 1 if side == 'white' else 0
            training_rows.append((history, player_turn, next_move))
    return training_rows

# Flatten the dataset into (X, y)
training_data = []
for _, row in df.iterrows():
    training_data.extend(generate_training_rows(row))

# Create dataframe
processed_df = pd.DataFrame(training_data, columns=["history", "turn", "target_move"])
```

---

### 🧠 2. Tokenize Moves (Input and Output)

```python
from collections import Counter

# Build move vocabulary
all_moves = [move for row in processed_df['history'] for move in row] + processed_df['target_move'].tolist()
vocab = sorted(set(all_moves))
move_to_idx = {move: i for i, move in enumerate(vocab)}
idx_to_move = {i: move for move, i in move_to_idx.items()}
vocab_size = len(vocab)

# Encode
processed_df['history_idx'] = processed_df['history'].apply(lambda seq: [move_to_idx[m] for m in seq])
processed_df['target_idx'] = processed_df['target_move'].apply(lambda m: move_to_idx[m])
```

## ✅ CNN Model in TensorFlow/Keras

```python
import tensorflow as tf
from tensorflow.keras import layers, models

model = models.Sequential([
    layers.Input(shape=(5, 1)),  # 4 moves + 1 turn
    layers.Conv1D(64, kernel_size=2, activation='relu'),
    layers.Conv1D(64, kernel_size=2, activation='relu'),
    layers.Flatten(),
    layers.Dense(128, activation='relu'),
    layers.Dropout(0.3),
    layers.Dense(vocab_size, activation='softmax')  # predict next move
])

model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])
model.summary()
```

## 🔮 Predicting Best Move

```python
def predict_next_move(model, move_history, turn):
    # Tokenize and pad
    move_indices = [move_to_idx[m] for m in move_history]
    padded = pad_sequences([move_indices], maxlen=4)
    padded = np.expand_dims(padded, -1)
    turn_tensor = np.array([[turn]])
    input_combined = np.concatenate([padded, np.expand_dims(turn_tensor, axis=-1)], axis=1)

    pred_probs = model.predict(input_combined)
    pred_idx = np.argmax(pred_probs)
    return idx_to_move[pred_idx]
```
